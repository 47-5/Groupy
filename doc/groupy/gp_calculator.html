 24    @staticmethod
 25    def Tm(group_number, parameters):
 26        """
 27        Calculating the freezing (melting) point. When user call Calculator.calculate_a_mol(), this method will be called automatically.
 28        In general, users do not need to call this function themselves.
 29        :param group_number: number of different groups in a given molecule.
 30        In general, users can use the result of groupy.gp_counter.Counter.count_a_mol()
 31        :param parameters: parameters used in group contribution method.
 32        In general, users can use internal data of groupy that can be loaded by groupy.gp_loader.Loader
 33        :return: float
 34        """
 35        right_side_eq = 0.00
 36        for i in group_number:  # 这样写是取了字典的键
 37            right_side_eq += group_number[i] * parameters[i]['Tm']
 38        Tm = parameters[1]['Tm0'] * log(max(right_side_eq, 1.0))   # 这里的1.0是为了上计算结果小于0K的都设置为0K
 39        return round(Tm, 3)

224    def delta_Hc(C_number, H_number, delta_Hf):
225        """
226        Calculating the enthalpy of combustion. When user call Calculator.calculate_a_mol(), this method will be called automatically if the molecule is a hydrocarbon compound.
227        In general, users do not need to call this function themselves.
228        Attention: This function was written with only hydrocarbons in mind, and the result of this function must be wrong when the molecule contains other heteroatoms (such as F, Cl, S...). When Calculator.calculate_a_mol() is called, Calculator.calculate_a_mol() will determine if the molecule to be calculated is a hydrocarbon molecule, and if not, it will automatically return None
229        :param C_number: number of different groups in a given molecule.
230        :param H_number: parameters used in group contribution method.
231        :param delta_Hf: Enthalpy of formation.
232        In general, users can use output of Calculator.delta_Hf().
233        :return: float
234        """
235        delta_Hc = -(-395.51 * C_number - 241.83 * H_number / 2 - delta_Hf)
236        return round(delta_Hc, 3)

250    def isp(C_number, H_number, q):
251        """
252        Calculating the Specific impulse(isp). When user call Calculator.calculate_a_mol(), this method will be called automatically if the molecule is a hydrocarbon compound.
253        In general, users do not need to call this function themselves.
254        Attention: This function was written with only hydrocarbons in mind, and the result of this function must be wrong when the molecule contains other heteroatoms (such as F, Cl, S...). When Calculator.calculate_a_mol() is called, Calculator.calculate_a_mol() will determine if the molecule to be calculated is a hydrocarbon molecule, and if not, it will automatically return None
255        :param C_number: number of different groups in a given molecule.
256        :param H_number: parameters used in group contribution method.
257        :param q: heat value.
258        In general, users can use output of Calculator.q() if the molecule is a hydrocarbon compound.
259        :return: float
260        """
261        H_C_ratio = H_number / C_number
262        parameter = q * (11.91 + H_C_ratio) / (43.66 + 8.936 * H_C_ratio)
263        if parameter < 0:  # todo这是无意中发现的错误，可能是因为算的不是碳氢分子（确实目前发现的出现错误的是含F原子的分子）
264            parameter = 0
265        isp = (2 * 0.556 * parameter) ** 0.5 / 9.8 * 1000
266        return round(isp, 3)

322        """
323        Calculating properties of a molecule.
324        :param mol: instance of rdkit.Chem.rdchem.Mol or SMILES str which will be converter to rdkit.Chem.rdchem.Mol automatically.
325        :param debug: bool. If set to be True, Calculator will print the results of Counter on screen. Default=False
326        :param check_hydrocarbon: bool. Since Calculator.delta_Hc() was designed for hydrocarbon, if set to True, Calculator will check whether the molecule is hydrocarbon.
327        If the molecule is not hydrocarbon, Calculator will not calculate delta_Hc, q and ISP. If set to False, Calculator will calculate these properties no matter whether the molecule is hydrocarbon.
328        Default=True.
329        :return: dict like {'smiles': init_smi,
330                    'molar_mass': value,
331                    'flash_point/K': value,
332                    'Tm/K': value, 'Tb/K': value, 'Tc/K': value,
333                    'Pc/bar': value, 'Vc/(cm3/mol)': value,
334                    'density/(g/cm3)': value,
335                    'delta_G/(KJ/mol)': value,
336                    'delta_Hf/(KJ/mol)': value,
337                    'delta_Hvap/(KJ/mol)': value,
338                    'delta_Hfus/(KJ/mol)': value,
339                    'molar_volume/(cm3/mol)(default298K)': value,
340                    'delta_Hc/(KJ/mol)': value,
341                    'mass_calorific_value_h/(MJ/kg)': value,
342                    'ISP': value,
343                    'note': value}
344        """
345        init_smi = mol
346        try:
347            if isinstance(mol, str):
348                mol = Chem.MolFromSmiles(mol)
349            group_number = self.counter.count_a_mol(mol, clear_mode=True, add_note=True)
350            if group_number.get('note', ''):
351                counter_note = group_number['note']
352                del group_number['note']
353            else:
354                counter_note = ''
355
356            if parameter_type == 'step_wise':
357                parameters = self.parameters_step_wise
358            elif parameter_type == 'simultaneous':
359                parameters = self.parameters_simultaneous
360            else:
361                raise NotImplemented('不可用的参数类型，只能使用step_wise或simultaneous')
362
363            if debug:

366            Tm = self.Tm(group_number=group_number, parameters=parameters)
367            Tb = self.Tb(group_number=group_number, parameters=parameters)
368            Tc = self.Tc(group_number=group_number, parameters=parameters)
369            Pc = self.Pc(group_number=group_number, parameters=parameters)
370            Vc = self.Vc(group_number=group_number, parameters=parameters)
371            delta_Gf = self.delta_Gf(group_number=group_number, parameters=parameters)
372            delta_Hf = self.delta_Hf(group_number=group_number, parameters=parameters)
373            delta_Hv = self.delta_Hv(group_number=group_number, parameters=parameters)
374            delta_Hfus = self.delta_Hfus(group_number=group_number, parameters=parameters)
375            C_number = self.C_number(mol)
376            H_number = self.H_number(mol)
377            molar_mass = self.molar_mass(mol)
378            flash_point = self.flash_point(group_number=group_number, parameters=parameters)
379            molar_volume = self.molar_volume(group_number=group_number, parameters=parameters)
380            density = self.density(molar_mass=molar_mass, Vm=molar_volume)
381            if check_hydrocarbon:
382                if self.is_hydrocarbon(mol):
383                    delta_Hc = self.delta_Hc(C_number=C_number, H_number=H_number, delta_Hf=delta_Hf)
384                    q = self.q(delta_Hc=delta_Hc, molar_mass=molar_mass)
385                    isp = self.isp(C_number=C_number, H_number=H_number, q=q)
386                else:
387                    delta_Hc = None
388                    q = None
389                    isp = None
390            else:
391                delta_Hc = self.delta_Hc(C_number=C_number, H_number=H_number, delta_Hf=delta_Hf)
392                q = self.q(delta_Hc=delta_Hc, molar_mass=molar_mass)
393                isp = self.isp(C_number=C_number, H_number=H_number, q=q)
394            smiles = self.smiles(mol)

402                    'density/(g/cm3)': density,
403                    'delta_G/(KJ/mol)': delta_Gf,
404                    'delta_Hf/(KJ/mol)': delta_Hf,
405                    'delta_Hvap/(KJ/mol)': delta_Hv,

407                    'molar_volume/(cm3/mol)(default298K)': molar_volume,

430    def calculate_mols(self, smiles_file_path, properties_file_path='gp_3x_result.csv', check_hydrocarbon=True, parameter_type='simultaneous'):
431        """
432        Calculating properties of a batch of molecules.
433        :param smiles_file_path: path of the file(.txt, .xlsx, .csv) in which saved SMILES.
434        :param properties_file_path: path of result file. default: 'gp_3x_result.csv'
435        :param check_hydrocarbon: bool. Since Calculator.delta_Hc() was designed for hydrocarbon, if set to True, Calculator will check whether the molecule is hydrocarbon.
436        If the molecule is not hydrocarbon, Calculator will not calculate delta_Hc, q and ISP. If set to False, Calculator will calculate these properties no matter whether the molecule is hydrocarbon.
437        Default=True.
438        :return: instance of pandas.DataFrame
439        """
440        print('reading input file...')
441        if smiles_file_path.endswith('.txt'):
442            smiles_iterator = list(open(smiles_file_path))
443        elif smiles_file_path.endswith('.xlsx'):
444            smiles_iterator = pd.read_excel(smiles_file_path)['smiles']
445        elif smiles_file_path.endswith('.csv'):
446            smiles_iterator = pd.read_csv(smiles_file_path)['smiles']
447        else:
448            print('无法识别的文件类型，请以.txt/.xlsx/.csv类型的文件作为输入。')
449            return None
450        mol_number = len(smiles_iterator)
451        print('reading completed，A total of {} molecules detected, start calculating properties...'.format(mol_number))
452        print('start calculating...')
453        properties_dict_list = []
454        error_smi = []
455        for i in tqdm(smiles_iterator):
456            try:
457                properties_dict_list.append(self.calculate_a_mol(i, check_hydrocarbon=check_hydrocarbon, parameter_type=parameter_type))
458            except:
459                error_smi.append(i)
460        print('calculation completed!')
461        print('start to export result to {} ...'.format(properties_file_path))
462        result = pd.DataFrame(properties_dict_list)
463        result.to_csv(properties_file_path, index_label='index')
464        with open('error.txt', 'w') as f:
465            for i in error_smi:
466                f.write(i + '\n')
467        print('Done!')
468        return result
